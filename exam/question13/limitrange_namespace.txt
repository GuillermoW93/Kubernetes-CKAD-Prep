Create a LimitRange that restricts max memory to 512Mi and max CPU to 1 for all pods.

apiVersion: v1
kind: LimitRange
metadata:
  name: limits
  namespace: your-namespace
spec:
  limits:
  - type: Container
    max:
      cpu: "1"
      memory: "512Mi"

kubectl get limitrange -n default
kubectl describe limitrange limits -n default

k apply -f pod2.yaml
